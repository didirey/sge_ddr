cadena_nombres = "".join(nombres).lower()

# 2. Iterar y contar, normalizando tildes a su base (a -> a, á -> a)
for caracter in cadena_nombres:
    if 'a' <= caracter <= 'z' or caracter in 'ñáéíóú':

            conteo_letras['a'] += 1
